function render(wx) { show('result'); // 즉시 결과 표시 const items = wx.items || []; const rainAmt = wx.rain || 0; const reh = items.find(i => i.category === 'REH')?.fcstValue || 70; let bg = '#15803d', lbl = '정상 시공 가능', msg = '안전 수칙을 준수하세요.'; // 1순위: 강수량 0.3mm (현철 님 마지노선) if (rainAmt >= 0.3) { bg = '#f97316'; lbl = "전일 강수량 확인"; msg = `새벽 5시까지 ${rainAmt}mm 비 예보. 바닥 건조 확인 필수!`; } // 2순위: 습도 75% (결로 방어) else if (parseInt(reh) >= 75) { bg = '#b45309'; lbl = "결로/이슬 주의"; msg = `${wx.isSpecial ? '해안/산간 지형: ' : ''}습도 ${reh}%로 바닥이 눅눅할 수 있음.`; } const banner = document.getElementById('banner'); banner.style.background = bg; document.getElementById('blabel').textContent = lbl; document.getElementById('bsub').textContent = msg; }

// index.html 파일 내의 render 함수 부분을 아래 내용으로 교체하세요. function render(wx) { const cfg = MAT[ST.mat]; const items = wx.items || []; // 1. 습도 75% 기준 및 지형 가중치 (현철 님 지시) const humidLimit = 75; const currentHumi = items.find(i => i.category === 'REH')?.fcstValue || 70; const highHumi = parseInt(currentHumi) >= humidLimit; // 2. 강수량 0.3mm 및 새벽 5시 기준 (현철 님 지시) const rainAmount = wx.rain || 0; let color = '#15803d', title = '', sub = ''; if (rainAmount >= 0.3) { color = '#f97316'; // 주황색 title = "전일 강수량 확인"; // 현철 님 지시 문구 sub = `새벽 5시까지 누적 ${rainAmount}mm 예보. 바닥 상태 확인 필수.`; } else if (highHumi) { color = '#b45309'; // 노란색 title = "결로/이슬 주의"; sub = `${wx.isSpecial ? '지형 특성상 ' : ''}습도 ${currentHumi}%로 바닥이 눅눅할 수 있음.`; } else { color = '#15803d'; // 초록색 title = "정상 작업 가능"; sub = "강수 및 결로 위험 낮음. 현장 안전 수칙 준수."; } // 배너 업데이트 const banner = document.getElementById('banner'); banner.style.background = color; document.getElementById('blabel').textContent = title; document.getElementById('bsub').textContent = sub; // 나머지 기존 렌더링 로직(이슬 건조 시간 등)은 그대로 유지됩니다. show('result'); }
